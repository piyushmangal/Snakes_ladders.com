import random class SnakesAndLadders: def __init__(self, players): self.board = self.create_board() self.players = players self.positions = {player: 0 for player in players} self.winner = None def create_board(self): board = list(range(1, 101)) snakes = {16: 6, 47: 26, 49: 11, 56: 53, 62: 19, 64: 60, 87: 24, 93: 73, 95: 75, 98: 78} ladders = {1: 38, 4: 14, 9: 31, 21: 42, 28: 84, 36: 44, 51: 67, 71: 91, 80: 100} for start, end in snakes.items(): board[start - 1] = end for start, end in ladders.items(): board[start - 1] = end return board def roll_dice(self): return random.randint(1, 6) def move_player(self, player): dice_roll = self.roll_dice() current_position = self.positions[player] new_position = current_position + dice_roll if new_position > 100: new_position = current_position else: new_position = self.board[new_position - 1] self.positions[player] = new_position print(f"{player} rolled a {dice_roll} and moved to {new_position}") if new_position == 100: self.winner = player def play(self): while not self.winner: for player in self.players: self.move_player(player) if self.winner: print(f"{self.winner} wins the game!") break if __name__ == "__main__": players = ["Alice", "Bob"] game = SnakesAndLadders(players) game.play()
